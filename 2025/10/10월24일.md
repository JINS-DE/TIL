## [백준 1135 뉴스전하기](https://www.acmicpc.net/problem/1135)
```python
"""
dp[u]는 u번째까지 걸리는 최소 시간
최소 시간 : 가장 많은 가지가 있는 애부터 처리해야 최소 시간 보장
   u
 / | \
dx dy dz

d1 = dx,dy,dz 중에 가장 큰 애
d2 = dx,dy,dz 중에 2번째로 그 다음 큰 애
d3 = dx,dy,dz 중에 3번째로 큰 애
dp[u] = max(d1+1,d2+2,d3+3)
"""

import sys
input = sys.stdin.readline
N = int(input())
parent = list(map(int,input().split()))
child = [[] for _ in range(N)]

for i in range(1,N):
    child[parent[i]].append(i)

dp = [0]*N
def dfs(u):
    # dp[u]의 값을 구하는게 목표

    child_lst = []
    for v in child[u]:
        dfs(v)
        child_lst.append(dp[v])
    
    child_lst.sort(reverse=True)

    dp[u]=0
    for i in range(len(child_lst)):
        dp[u] = max(dp[u],i+1+child_lst[i])

dfs(0)
print(dp[0])
```
